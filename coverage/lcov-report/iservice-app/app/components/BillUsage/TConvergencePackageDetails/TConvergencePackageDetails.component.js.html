import LinearGradient from 'react-native-linear-gradient';
import PackageDetailHeader from '../BillUsage/BillUsageShared/PackageDetailHeader/PackageDetailHeader.component';
import Proptypes from 'prop-types';
import React, {Component} from 'react';
import result from 'lodash/result';
import styles from './TConvergencePackageDetails.styles';
import TMPackageDetailsExtra from '../TMPackageDetailsExtra/TMPackageDetailsExtra.component';
import TOLPackageDetail from '../TOLPackageDetails/TOLPackageDetails.component';
import TVPackageDetail from '../TVPackageDetails/TVPackageDetails.component';
import {BottomCaretView, ISText} from '../../Shared';
import {colors, fonts} from '../../../themes/constants.styles';
import {ScrollView, View} from 'react-native';
import {translate} from '../../../language/i18n/helper';
 
export class TConvergencePackageDetails extends Component {
  render () {
    const {TOLPackageDetails, TVPackageDetails, TMPackageDetails, userDetails, billCycle} = this.props;
    return (
      <View style={styles.container}>
        <BottomCaretView caretColor={colors.SECONDARY_TSMARTCHOICE_MIDDLE} iconSize = {fonts.FONT_SIZE_XXXL}>
          <LinearGradient colors={[colors.SECONDARY_TSMARTCHOICE_START, colors.SECONDARY_TSMARTCHOICE_MIDDLE, colors.SECONDARY_TSMARTCHOICE_END]}
            start={{x: 0, y: 1}} end={{x: 1, y: 1}} locations={[0, 0.33, 1]} style={styles.headerContainer}>
            <PackageDetailHeader endDate ={result(billCycle, 'code', 0)} name = {translate('BILLS_USAGE_TRUESMARTCHOICE')} titleStyle={styles.titleStyle} leftSectionStyle={styles.leftSectionStyle}/>
          </LinearGradient>
        </BottomCaretView>
        <ISText style={styles.packageTitle} type='BOLD'>{translate('BILLS_USAGE_TRUESMARTCHOICE')}</ISText>
        <ScrollView style={styles.scrollContainer}>
          <TMPackageDetailsExtra packageDetailExtra={TMPackageDetails} isConvergence/>
          <TOLPackageDetail packageDetails = {TOLPackageDetails} isConvergence userDetails = {userDetails}/>
          <TVPackageDetail packageDetails = {TVPackageDetails} isConvergence userDetails = {userDetails}/>
        </ScrollView>
      </View>
    );
  }
}
 
TConvergencePackageDetails.defaultProps = {
  billCycle: {}
};
 
TConvergencePackageDetails.propTypes = {
  TOLPackageDetails: Proptypes.object,
  TVPackageDetails: Proptypes.object,
  TMPackageDetails: Proptypes.object,
  userDetails: Proptypes.object,
  billCycle: Proptypes.object
};
 
export default TConvergencePackageDetails;
