import bannerImage from '../../../../../assets/bill_usage_banner.jpg';
import BaseProductCollapsibleBar from '../BaseProductCollapsibleBar/BaseProductCollapsibleBar.component';
import groupBy from 'lodash/groupBy';
import noop from 'lodash/noop';
import Proptypes from 'prop-types';
import React, {Component} from 'react';
import styles from './TSmartChoiceCard.style';
import {BillHistoryList, CurrentUsageCard, ProductCardTitle} from '../../BillUsageShared';
import {CaretTabs} from '../../../../../components/Shared';
import {postpaidData} from '../../../mockData';
import {translate} from '../../../../../language/i18n/helper';
import {View} from 'react-native';
 
const calculateTotal = (value1, value2, precision = 2) => {
  const multiplier = Math.pow(10, precision);
  const val1 = value1 * multiplier;
  const val2 = value2 * multiplier;
  const total = (val1 + val2) / multiplier;
  return total;
};
 
class TSmartChoiceCard extends Component {
 
  calculateDueBills = (bills) => {
    if (!bills) {
      return null;
    }
    const dueBills = groupBy(bills, 'invoiceCycle');
    return Object.keys(dueBills).map((cycleDate) => {
      let response;
      const invoices = dueBills[cycleDate];
      invoices.forEach((invoice) => {
        if (response) {
          response.billAmount = calculateTotal(response.billAmount, invoice.billAmount);
          response.unpaidAmount = calculateTotal(response.unpaidAmount, invoice.unpaidAmount);
        } else {
          response = {...invoice};
        }
      });
      return response;
    });
  };
 
  state = {
    selectedTabIndex: 0,
    isCollapsed: this.props.isCollapsed,
    dueBills: this.calculateDueBills(this.props.dueBills)
  }
 
  componentWillReceiveProps ({dueBills}) {
    if (this.props.dueBills !== dueBills) {
      this.setState({dueBills: this.calculateDueBills(dueBills)});
    }
  }
 
  onTabPress = (selectedTabIndex) => {
    if (selectedTabIndex === 1 && !this.state.dueBills) {
      this.props.getDueBills();
    }
    this.setState({selectedTabIndex});
  }
 
  onCollapseToggle = () => {
    this.setState({isCollapsed: !this.state.isCollapsed});
  }
 
  render () {
    const {amount, plan, subtitle, /* TODO productDetail,*/ status, onChecked, isChecked} = this.props;
    const {dueBills, isCollapsed} = this.state;
    const selectedTabNumber = (this.state.selectedTabIndex === 1 && dueBills) ? 1 : 0;
    const headerList = [{text: translate('BILLS_USAGE_CURRENT_USAGE')}, {text: translate('BILLS_USAGE_BILL_DETAILS')}];
    return (
      <BaseProductCollapsibleBar status={status} type={'SMARTCHOICE'} leftTitleText={translate('BILLS_USAGE_TRUESMARTCHOICE')} amountText={amount} isCollapsed={isCollapsed} onCollapseToggle={this.onCollapseToggle} onChecked={onChecked} isChecked={isChecked}>
        <View style={styles.container}>
          <View style={styles.productTitle}>
            <ProductCardTitle text={plan} subtext={subtitle}/>
          </View>
          <CaretTabs onTabPress={this.onTabPress} selectedIndex={selectedTabNumber} headerList={headerList} activeTabStyle={styles.activeTabStyle} bodyStyle={styles.tabContainer}>
            {selectedTabNumber === 0 && <CurrentUsageCard data={/* TODO productDetail.currentUsage || */postpaidData[1].currentUsage} bannerImage={bannerImage} textStyle={styles.usageText}/>}
            {selectedTabNumber === 1 && <BillHistoryList dueBills={dueBills} onBillToggle={this.onBillToggle} />}
          </CaretTabs>
        </View>
      </BaseProductCollapsibleBar>
    );
  }
}
 
TSmartChoiceCard.defaultProps = {
  amount: 0,
  plan: '',
  subtitle: '',
  productDetail: {},
  status: 'ACTIVE',
  isCollapsed: true,
  onChecked: noop,
  isChecked: false,
  getDueBills: noop
};
 
TSmartChoiceCard.propTypes = {
  amount: Proptypes.number,
  plan: Proptypes.string,
  subtitle: Proptypes.string,
  dueBills: Proptypes.array,
  /* TODO productDetail: Proptypes.object,*/
  status: Proptypes.string,
  isCollapsed: Proptypes.bool,
  onChecked: Proptypes.func,
  isChecked: Proptypes.bool,
  getDueBills: Proptypes.func
};
 
export default TSmartChoiceCard;
