import HeaderWithCaret from '../BillUsage/BillUsageShared/HeaderWithCaret.component';
import moment from '../../../utils/moment.util';
import partition from 'lodash/partition';
import PropTypes from 'prop-types';
import React, {Component} from 'react';
import result from 'lodash/result';
import styles from './TMExtraPackage.style';
import {bytesToSize} from '../../../utils/common.util';
import {colors} from '../../../themes/constants.styles';
import {ICON_MAP, QUOTA_TYPE, QUOTA_VALUE} from '../../../config/billUsage.config';
import {ISText, ProgressBar, WrapperCard} from '../../Shared';
import {ScrollView, View} from 'react-native';
import {translate} from '../../../language/i18n/helper';
 
class TMExtraPackage extends Component {
 
  render () {
    const {extraPackage, mainPackage, isPrepaid} = this.props;
    const filteredExtraPackages = partition(extraPackage, (item) => !item.isActive);
    const filteredMainPackage = partition(mainPackage, (item) => !item.isActive);
    const rowContent = (packageRowItem, index, packageType) => {
      const {quotaType, packageCode, remainingQuota, quotaUnit, initialQuota, endDate} = packageRowItem;
      const title = result(ICON_MAP[quotaType], 'titleText', '');
      const subTitle = ' (' + packageType + ')' + ' : ' + packageCode;
      const remainingValue = Math.round(parseInt(remainingQuota));
      const initialValue = Math.round(parseInt(initialQuota));
      const isUnlimited = initialQuota === QUOTA_VALUE.UNLIMITED || remainingQuota === QUOTA_VALUE.UNLIMITED;
      // For prepaid: display remaining value ; For postpaid: display consumed value
      const value = (isPrepaid) ? remainingValue : (initialValue - remainingValue);
      const unit = (quotaUnit === QUOTA_VALUE.MINUTES) ? translate('BILLS_USAGE_MIN') : quotaUnit;
      const percentageValue = (isUnlimited || !initialQuota) ? 100 : (value * 100) / initialValue;
      let initialValueDisplay = initialValue;
      let valueDisplay = value;
      let unitDisplay = unit;
      // For (QUOTA_TYPE === DATA): 1. Size initial value 2. Size remaining value based on initial value sized unit
      if (quotaType === QUOTA_TYPE.DATA && !isUnlimited) {
        const intialSizedValue = (initialQuota) ? bytesToSize(initialValueDisplay, unit) : {}; // bytesToSize: value, currentUnit, targetUnit
        initialValueDisplay = intialSizedValue.value || '';
        const valueToBeDisplay = bytesToSize(valueDisplay, unit, intialSizedValue.unit || '');
        valueDisplay = valueToBeDisplay.value;
        unitDisplay = valueToBeDisplay.unit;
      }
 
      return (
        <View key={index} style={styles.packageItem}>
          <View style={styles.packageItemRow}>
            <View style={styles.packageRowLeft}>
              <ISText type='BOLD' style={styles.itemTitle}>{title ? translate(title) : quotaType}
                <ISText type='BOLD' style={styles.itemSubtitle}>{subTitle}</ISText>
              </ISText>
            </View>
            <View style={styles.packageRowRight}>
              <ISText type='BOLD' style={(isUnlimited) ? styles.unlimitedText : styles.consumedValue}>{(!isUnlimited) ? valueDisplay : translate('BILLS_USAGE_UNLIMITED')}
                {(!isUnlimited && unit) && <ISText type='BOLD' style={styles.consumedValueUnit}>{' ' + unitDisplay}</ISText>}
              </ISText>
            </View>
          </View>
          <View style={styles.packageItemRow}>
            <View style={styles.packageRowLeft}>
              <ProgressBar fillColor={colors.SECONDARY_TMOVE} backgroundColor={colors.PRIMARY_BG_SEPARATOR} percentage={percentageValue} />
            </View>
            <View style={styles.packageRowRight}>
              {(!isUnlimited && unit && initialQuota) && <ISText type='SEMIBOLD' style={styles.totalValue}>{translate('BILLS_USAGE_OF') + ' ' + initialValueDisplay} {unitDisplay}</ISText>}
            </View>
          </View>
          {endDate ? <ISText type='SEMIBOLD' style={styles.validityText}>{translate('BILLS_USAGE_VALID_UNTIL') + ' ' + new moment(endDate).format('DD/MM/YYYY')} </ISText> : null}
        </View>
      );
    };
 
    const activeMainPackageContent = filteredMainPackage[0].map((packageRowItem, index) => rowContent(packageRowItem, index, translate('BILLS_USAGE_MAIN')));
    const activeExtraPackageContent = filteredExtraPackages[0].map((packageRowItem, index) => rowContent(packageRowItem, index, translate('BILLS_USAGE_EXTRA')));
    const inactiveMainPackageContent = filteredMainPackage[1].map((packageRowItem, index) => rowContent(packageRowItem, index, translate('BILLS_USAGE_MAIN')));
    const inactiveExtraPackageContent = filteredExtraPackages[1].map((packageRowItem, index) => rowContent(packageRowItem, index, translate('BILLS_USAGE_EXTRA')));
    return (
      <ScrollView style={styles.scrollContainer} contentContainerStyle={styles.scrollContentContainer}>
        <WrapperCard>
          <HeaderWithCaret headerColor={colors.PRIMARY_SUBMENU} caretShadow={true}>
            <ISText type='BOLD' style={styles.cardHeaderText}>{translate('EXTRA_PACKAGE_ACTIVE')}</ISText>
          </HeaderWithCaret>
          <View style={styles.cardContent}>
            {activeMainPackageContent}
            {activeExtraPackageContent}
          </View>
        </WrapperCard>
        <WrapperCard>
          <HeaderWithCaret headerColor={colors.PRIMARY_DISABLED_BG_TEXT} caretShadow={true}>
            <ISText type='BOLD' style={styles.cardHeaderText}>{translate('EXTRA_PACKAGE_INACTIVE')}</ISText>
          </HeaderWithCaret>
          <View style={styles.cardContent}>
            {inactiveMainPackageContent}
            {inactiveExtraPackageContent}
          </View>
        </WrapperCard>
      </ScrollView>
    );
  }
}
 
TMExtraPackage.defaultProps = {
  extraPackage: [],
  mainPackage: [],
  isPrepaid: false
};
 
TMExtraPackage.propTypes = {
  extraPackage: PropTypes.array,
  mainPackage: PropTypes.array,
  isPrepaid: PropTypes.bool
};
 
export default TMExtraPackage;
