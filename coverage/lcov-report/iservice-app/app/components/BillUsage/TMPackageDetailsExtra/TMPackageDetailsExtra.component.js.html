import HeaderWithCaret from '../BillUsage/BillUsageShared/HeaderWithCaret.component';
import PackageDetailHeader from '../BillUsage/BillUsageShared/PackageDetailHeader/PackageDetailHeader.component';
import PropTypes from 'prop-types';
import React, {Component} from 'react';
import styles from './TMPackageDetailsExtra.style';
import TMExtraPackage from './TMExtraPackage.component';
import TMPackageDetails from './TMPackageDetails.component';
import {CaretTabs, WrapperCard} from '../../Shared';
import {translate} from '../../../language/i18n/helper';
import {View} from 'react-native';
 
class TMPackageDetailsExtra extends Component {
  state = {
    selectedTabIndex: 0
  }
 
  onTabPress = (selectedTabIndex) => {
    this.setState({selectedTabIndex});
  }
  render () {
    const {pricePlanInfo, isPrepaid, mainPackage,  extraPackage, billCycle, creditLimit, offerDesc, phoneNumber, isConvergence} = this.props;
    const {selectedTabIndex} = this.state;
    const headerList = [{
      text: translate('PACKAGE_DETAIL__TITLE')
    }, {
      text: translate('EXTRA_PACKAGE_TITLE')
    }];
    const tmPackageDetail = <View style={styles.container}>
      <CaretTabs onTabPress={this.onTabPress} selectedIndex={selectedTabIndex} headerList={headerList} headerContainerStyle={styles.caretTabContainer} bodyStyle={styles.bodyStyle}>
        {selectedTabIndex === 0 && <TMPackageDetails mainPackage={mainPackage} pricePlanInfo={pricePlanInfo} billCycle={billCycle} creditLimit={creditLimit} offerDesc={offerDesc} isPrepaid={isPrepaid} phoneNumber={phoneNumber}/>}
        {selectedTabIndex === 1 && <TMExtraPackage mainPackage={mainPackage} extraPackage={extraPackage} isPrepaid={isPrepaid}/>}
      </CaretTabs>
    </View>;
    return (
      isConvergence ?
        <WrapperCard contentContainerStyles={styles.wrapperCardStyle}>
          <HeaderWithCaret headerColor = {styles.headerColor} style = {styles.headerStyle}>
            <PackageDetailHeader isConvergence = {isConvergence} phoneNumber = {phoneNumber} name = {translate('BILLS_USAGE_TRUEMOVE')}/>
          </HeaderWithCaret>
          {tmPackageDetail}
        </WrapperCard>
        : tmPackageDetail
    );
  }
}
 
TMPackageDetailsExtra.defaultProps = {
  pricePlanInfo: {},
  mainPackage: [],
  extraPackage: [],
  billCycle: {},
  creditLimit: '',
  offerDesc: '',
  isPrepaid: false,
  phoneNumber: '',
  isConvergence: false
};
 
TMPackageDetailsExtra.propTypes = {
  pricePlanInfo: PropTypes.object,
  mainPackage: PropTypes.array,
  extraPackage: PropTypes.array,
  billCycle: PropTypes.object,
  creditLimit: PropTypes.string,
  offerDesc: PropTypes.string,
  isPrepaid: PropTypes.bool,
  phoneNumber: PropTypes.string,
  isConvergence: PropTypes.bool
};
 
export default TMPackageDetailsExtra;
