// -------disabling for camelcase rule----------  TODO: fix when start implementing UI

import PropTypes from 'prop-types';
import React, {Component} from 'react';
import result from 'lodash/result';
import {bindActionCreators} from 'redux';
import {Button, Checkbox, IconTextInput, ISText} from '../../components/Shared';
import {connect} from 'react-redux';
import {payNewCustomer, payUsingExistingCard, removeExisingCard} from '../../redux/actions/index.actions';
import {ScrollView, View} from 'react-native';
 
class PaymentMethods extends Component {
  state = {
    form: {
      name: 'JOHN DOE',
      city: 'Bangkok',
      postal_code: '10320',
      number: '4242424242424242',
      security_code: '123',
      expiration_month: '10',
      expiration_year: '2019',
      saveCard: false
    },
    savedCards: result(this.props, 'navigation.state.params.savedCards', [])
  }
  setFormVal = (keyToChange) => (newText) => {
    const newFormState = {...this.state.form, [keyToChange]: newText};
    this.setState({form: newFormState});
  }
  onFormSubmit = () => this.props.addCard({
    ...this.state.form, amount: result(this.props, 'navigation.state.params.amount', '')
  })
 
  payByCard = (id) => () => {
    const payload = {cardID: id, amount: result(this.props, 'navigation.state.params.amount', '')};
    this.props.payByCard(payload);
  }
 
  deleteCard = (id) => () => {
    this.props.deleteCard(id);
    this.setState({savedCards: this.state.savedCards.filter((card) => card.id !== id)});
  }
 
  creditcardForm = () => (
    <View>
      <ISText>Cardholder Name</ISText>
      <IconTextInput iconName={'pay-for-others'} defaultValue={this.state.form.name} placeholder={'Cardholder Name'} onChangeText={this.setFormVal('name')} />
      <ISText>Card Number</ISText>
      <IconTextInput iconName={'pay-for-others'} defaultValue={this.state.form.number} placeholder={'Card Number'} onChangeText={this.setFormVal('number')} />
      <ISText>City</ISText>
      <IconTextInput iconName={'pay-for-others'} defaultValue={this.state.form.city} placeholder={'City'} onChangeText={this.setFormVal('city')} />
      <ISText>Postal Code</ISText>
      <IconTextInput iconName={'pay-for-others'} defaultValue={this.state.form.postal_code} placeholder={'Postal Code'} onChangeText={this.setFormVal('postal_code')} />
      <ISText>Security Code</ISText>
      <IconTextInput iconName={'pay-for-others'} defaultValue={this.state.form.security_code} placeholder={'Security Code'} onChangeText={this.setFormVal('security_code')} />
      <ISText>Expiration Month</ISText>
      <IconTextInput iconName={'pay-for-others'} defaultValue={this.state.form.expiration_month} placeholder={'Expiration Month'} onChangeText={this.setFormVal('expiration_month')} />
      <ISText>Expiration Year</ISText>
      <IconTextInput iconName={'pay-for-others'} defaultValue={this.state.form.expiration_year} placeholder={'Expiration Year'} onChangeText={this.setFormVal('expiration_year')} />
      <ISText>Save Card</ISText>
      <Checkbox onChange={this.setFormVal('saveCard')} isChecked={false}/>
      <Button text='Pay' onPress={this.onFormSubmit}/>
    </View>)
 
  creditCardDetails = ({brand, expiration_month, expiration_year, name, id, last_digits}, i) => (
    <View key={i}>
      <ISText>Brand: {brand}</ISText>
      <ISText>Expiry: {expiration_month + '/' + expiration_year}</ISText>
      <ISText>name: {name}</ISText>
      <ISText>id: {id}</ISText>
      <ISText>Card No: **** **** **** {last_digits}</ISText>
      <View>
        <Button type='inline' text='Pay using this card' onPress={this.payByCard(id)}/>
        <Button type='inline' text='Delete Card' onPress={this.deleteCard(id)}/>
      </View>
    </View>)
 
  render () {
    const {savedCards} = this.state;
    const isCardsPresent = savedCards.length > 0;
    return (
      <ScrollView>
        <ISText>
          {'\n'}{isCardsPresent ? 'Pick one of the saved cards above' : 'No saved cards. Please add a new one'}{'\n'}
        </ISText>
        {isCardsPresent ? savedCards.map(this.creditCardDetails) : this.creditcardForm()}
      </ScrollView>
    );
  }
}
 
PaymentMethods.propTypes = {
  addCard: PropTypes.func,
  deleteCard: PropTypes.func,
  payByCard: PropTypes.func
};
export const mapDispatchToProps = (dispatch) => ({
  addCard: bindActionCreators(payNewCustomer, dispatch),
  payByCard: bindActionCreators(payUsingExistingCard, dispatch),
  deleteCard: bindActionCreators(removeExisingCard, dispatch)
});
export default connect(null, mapDispatchToProps)(PaymentMethods);
