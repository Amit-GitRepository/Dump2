/* Payment: dummy payment page. This file is just to show end to end payment.
            Designs are not finalised. This is just a wireframe
*/
 
import PropTypes from 'prop-types';
import React, {Component} from 'react';
import {Button, IconTextInput, ISText} from '../../components/Shared';
import {connect} from 'react-redux';
import {showPaymentMethods} from '../../redux/actions/index.actions';
import {View} from 'react-native';
 
// No separate file because it won't be used in future
const styles = {
  container: {
    padding: 20,
    justifyContent: 'space-between',
    flex: 1
  }
};
 
class Payment extends Component {
  state = {amount: ''}
  onAmountChange = (amount) => this.setState({amount})
  render () {
    return (
      <View style={styles.container}>
        <View>
          <ISText>Amount(in thb subunits){'\n'}</ISText>
          <IconTextInput iconName={'pay-for-others'} placeholder={'Amount'} onChangeText={this.onAmountChange} />
        </View>
        <View>
          <Button text='Next' onPress={this.props.onNextPress(this.state.amount)} />
        </View>
      </View>);
  }
}
Payment.propTypes = {
  onNextPress: PropTypes.func
};
 
const mapDispatchToProps = (dispatch) => ({
  onNextPress: (amount) => () => dispatch(showPaymentMethods(amount))
});
const ConnectedPayment = connect(null, mapDispatchToProps)(Payment);
export default ConnectedPayment;
