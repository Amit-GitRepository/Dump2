import _ from 'lodash';
import BillingMethods from '../../components/BillingMethods/BillingMethods/BillingMethods.component';
import ComponentWithTab from '../../components/ComponentWithTab/ComponentWithTab.component';
import PropTypes from 'prop-types';
import React from 'react';
import {connect} from 'react-redux';
import {NavigationActions} from 'react-navigation';
import {View} from 'react-native';
import * as actions from '../../redux/actions/index.actions';
 
function getAccountIds (productList) {
  const accountIds = [];
  _.forEach(productList, (array, key) => {
    if (key === 'tmhPrepaid') {
      return;
    }
 
    if (key === 'conv') {
      // so for convergence account - All the accounts for which
      // convergenceCode is same && isBillConsolidated
      // then use any one of the accountIds
      // otherwise use all the accountIds individually
      _.map(array, (val) => {
        const productList = _.result(val, 'products', []);
        if (_.every(productList, (item) => item.isBillConsolidated)) {
          accountIds.push(productList[0].accountId);
        } else {
          _.map(productList, (item) => {
            accountIds.push(item.accountId);
          });
        }
      });
    }
    _.map(array, (val) => {
      accountIds.push(val.accountId);
    });
  });
  return accountIds;
}
 
class BillingMethodsPage extends ComponentWithTab {
  state = {
    isBPFetched: false
  }
 
  componentWillMount () {
    this.props.getBillingPreferences(getAccountIds(this.props.productList).toString());
  }
 
  componentWillReceiveProps (newProps) {
    const {productList} = newProps;
    const {isBPFetched} = this.state;
    if (!_.every(productList, _.isEmpty) && !isBPFetched) {
      this.props.getBillingPreferences(getAccountIds(productList).toString());
      this.setState({isBPFetched: true});
    }
  }
 
  navigateToScreen = (screen) => (params = {}) => {
    this.props.goToScreen(screen, params);
  }
 
  renderWithTab () {
    const {billPreferences} = this.props;
    return (
      <View>
        <BillingMethods
          ebillContent={_.union(billPreferences.email || [], billPreferences.sms || [])}
          paperBillContent={billPreferences.paper || []}
          onApplyEbillAction={this.navigateToScreen('EbillAdd')}
          onEditEbillAction={this.navigateToScreen('EbillEdit')}
        />
      </View>
    );
  }
}
 
BillingMethodsPage.propTypes = {
  goToScreen: PropTypes.func,
  productList: PropTypes.object,
  billPreferences: PropTypes.object,
  getBillingPreferences: PropTypes.func
};
 
const mapStateToProps = (state) => ({
  productList: _.result(state, 'billUsage.productList', {}),
  billPreferences: _.result(state, 'ebill.billPreferences', {})
});
 
export const mapDispatchToProps = (dispatch) => ({
  goToScreen: (screenName, params) => dispatch(NavigationActions.navigate({routeName: screenName, params})),
  getBillingPreferences: (accountIds) => dispatch(actions.getBillingPreferences(accountIds))
});
 
export default connect(mapStateToProps, mapDispatchToProps)(BillingMethodsPage);
