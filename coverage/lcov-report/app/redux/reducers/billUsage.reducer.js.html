import result from 'lodash/result';
import {SET_BILL_USAGE_PRODUCT_BILL_DETAIL, SET_BILL_USAGE_PRODUCT_COLLAPSE_STATUS, SET_BILL_USAGE_PRODUCT_DETAIL, SET_BILL_USAGE_PRODUCT_LIST, TOGGLE_BILL_USAGE_PRODUCT_CHECK_STATUS} from '../actions/index.actions';
 
const initialState = {
  productList: {
    'ccbsList': [], // postpaid
    'trueList': [], // true online
    'iccList': [],  // true vision
    'prepayList': [] // prepaid
  },
  productDetail: {
    'ccbsList': {},
    'trueList': {},
    'iccList': {},
    'prepayList': {}
  },
  billDetail: {
    'ccbsList': {},
    'trueList': {},
    'iccList': {},
    'prepayList': {}
  }
};
 
const addAttributesToProducts = (productList) => {
  const list = productList.map((eachProduct) => ({
    ...eachProduct,
    arBalance: parseFloat(eachProduct.arBalance) || 0.0,
    isChecked: false,
    isCollapsed: true
  }));
  return list;
};
 
export default function billUsage (state = initialState, action) {
  switch (action.type) {
  case SET_BILL_USAGE_PRODUCT_LIST : {
    const ccbsList = addAttributesToProducts(result(action, 'payload.ccbsList', []));
    const trueList = addAttributesToProducts(result(action, 'payload.trueList', []));
    const iccList = addAttributesToProducts(result(action, 'payload.iccList', []));
    const prepayList = addAttributesToProducts(result(action, 'payload.prepayList', []));
    return {
      ...state,
      productList: {
        'ccbsList': ccbsList,
        'trueList': trueList,
        'iccList': iccList,
        'prepayList': prepayList
      }
    };
  }
 
  case SET_BILL_USAGE_PRODUCT_COLLAPSE_STATUS: {
    const {productType, subscriberId, isCollapsed} = action.payload;
    const productGroupList = state.productList[productType] || [];
    const updatedProductGroupList = productGroupList.map((eachProduct) => {
      if (eachProduct.subscriberId === subscriberId) {
        return {...eachProduct, isCollapsed};
      }
      return eachProduct;
    });
    return {...state, productList: {...state.productList, [productType]: updatedProductGroupList}};
  }
 
  case TOGGLE_BILL_USAGE_PRODUCT_CHECK_STATUS: {
    const {productType, subscriberId} = action.payload;
    const productGroupList = state.productList[productType] || [];
    const updatedProductGroupList = productGroupList.map((eachProduct) => {
      if (productType === 'prepayList') {
        eachProduct.isChecked = false;
      }
      if (eachProduct.subscriberId === subscriberId) {
        eachProduct.isChecked = !eachProduct.isChecked;
      }
      return {...eachProduct};
    });
    return {...state, productList: {...state.productList, [productType]: updatedProductGroupList}};
  }
 
  case SET_BILL_USAGE_PRODUCT_DETAIL: {
    const {productType, subscriberId, data} = action.payload;
    const productGroup = state.productDetail[productType] || {};
    productGroup[subscriberId] = data;
    return {...state, productDetail: {...state.productDetail, [productType]: productGroup}};
  }
 
  case SET_BILL_USAGE_PRODUCT_BILL_DETAIL: {
    const {productType, subscriberId, data} = action.payload;
    const productGroup = state.billDetail[productType] || {};
    productGroup[subscriberId] = data;
    return {...state, billDetail: {...state.billDetail, [productType]: productGroup}};
  }
  default :
    return state;
  }
}
