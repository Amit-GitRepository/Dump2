import Button from '../Shared/Button/Button.component';
import ISText from '../Shared/ISText/ISText.component';
import Modal from '../Shared/Modal/Modal.component';
import noop from 'lodash/noop';
import PropTypes from 'prop-types';
import React, {Component} from 'react';
import styles from './LoginPrompt.style';
import {translate} from '../../language/i18n/helper';
import {View} from 'react-native';
 
class LoginPrompt extends Component {
  state = {
    toRequestLogin: false
  };
 
  closeLoginPrompt = () => {
    this.setState({toRequestLogin: false});
    this.props.toggleLoginPrompt();
  }
 
  loginRequest = () => {
    this.setState({toRequestLogin: true});
    this.props.toggleLoginPrompt(this.props.nextLoginAction);
  }
 
  onLoginRequest = () => {
    if (this.state.toRequestLogin) {
      this.props.loginRequest(this.props.nextLoginAction);
    }
  }
 
  render () {
    const {showPrompt} = this.props;
    // Do not have designs, translations currently, to be updated
    return (
      <Modal isVisible={showPrompt} animationIn={'zoomIn'} animationOut={'zoomOut'}
        animationInTiming={500} backdropTransitionInTiming={1000} backdropTransitionOutTiming={1000}
        animationOutTiming={500} onModalHide={this.onLoginRequest}>
        <View style={styles.container}>
          <ISText>{translate('SIDE_MENU__LOGIN_PROMPT_NEED_ACCESS')}</ISText>
          <View style={styles.buttonContainer}>
            <Button onPress={this.closeLoginPrompt} text={translate('SIDE_MENU__LOGIN_PROMPT_CANCEL')}/>
            <Button onPress={this.loginRequest} text={translate('SIDE_MENU__LOGIN_PROMPT_LOGIN')}/>
          </View>
        </View>
      </Modal>
    );
  }
}
 
LoginPrompt.defaultProps = {
  showPrompt: false,
  toggleLoginPrompt: noop,
  loginRequest: noop
};
 
LoginPrompt.propTypes = {
  showPrompt: PropTypes.bool,
  toggleLoginPrompt: PropTypes.func,
  nextLoginAction: PropTypes.object,
  loginRequest: PropTypes.func
};
 
export default LoginPrompt;
