import Icon from '../Icon/Icon.component';
import ISText from '../../Shared/ISText/ISText.component';
import PropTypes from 'prop-types';
import React, {Component} from 'react';
import styles from './SquareIconButton.style';
import Touchable from '../Touchable/Touchable.component';
import {View} from 'react-native';
 
class SquareIconButton extends Component {
  render () {
    const {title, subtitle, iconName} = this.props;
    return (
      <Touchable style={styles.container}>
        <View style={styles.wrapper}>
          <Icon name={iconName} style={styles.icon}/>
          <ISText style={styles.title} type='BOLD' i18nKey={title} />
          <View style={styles.subtextContainer}>
            <Icon style={styles.subIcon} size={10} transform={styles.subIcon.transform} name='chevron-right' />
            <ISText type={'SEMIBOLD'} style={styles.subtitle} noOfLines={1} ellipsizeMode={'tail'} i18nKey= {subtitle} />
          </View>
        </View>
      </Touchable>
    );
  }
}
 
SquareIconButton.defaultProps = {
  title: '',
  subtitle: '',
  iconName: null
};
 
SquareIconButton.propTypes = {
  title: PropTypes.string,
  subtitle: PropTypes.string,
  iconName: PropTypes.string
};
 
export default SquareIconButton;
