import PropTypes from 'prop-types';
import React, {Component} from 'react';
import resolveAssetSource from 'resolveAssetSource';
import styles from './Banner.styles';
import {ImageBackground} from 'react-native';
 
class Banner extends Component {
  state = {
    viewWidth: null
  }
 
  calculateStyle = () => {
    const {source, style} = this.props;
    const imageProps = resolveAssetSource(source);
    const heightStyle = {height: (imageProps.height / imageProps.width) * this.state.viewWidth};
    return {...styles.image, ...heightStyle, ...style};
  }
 
  onLayout = (event) => {
    const {width} = event.nativeEvent.layout;
    this.setState({viewWidth: width});
  }
 
  render () {
    const {source, ...extraProps} = this.props;
    const imageStyle = this.calculateStyle();
    return (
      <ImageBackground onLayout={this.onLayout} resizeMode={'contain'} {...extraProps} style={imageStyle} source={source}>
        {this.props.children}
      </ImageBackground>
    );
  }
}
 
Banner.defaultProps = {
  style: {},
  source: 0,
  isBackground: false
};
 
Banner.propTypes = {
  style: PropTypes.object,
  source: PropTypes.number,
  isBackground: PropTypes.bool,
  children: PropTypes.node
};
 
export default Banner;
