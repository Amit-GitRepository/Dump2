import DetailCardRow from '../BillUsage/BillUsageShared/DetailCardRow.component';
import HeaderWithCaret from '../BillUsage/BillUsageShared/HeaderWithCaret.component';
import Icon from '../../Shared/Icon/Icon.component';
import ISText from '../../Shared/ISText/ISText.component';
import PackageDetailHeader from '../BillUsage/BillUsageShared/PackageDetailHeader/PackageDetailHeader.component';
import PackageOverview from '../BillUsage/BillUsageShared/PackageOverview/PackageOverview.component';
import PropTypes from 'prop-types';
import React, {Component} from 'react';
import styles from './TMPackageDetails.style';
import WrapperCard from '../BillUsage/BillUsageShared/WrapperCard.component';
import {ScrollView, View} from 'react-native';
import {translate} from '../../../language/i18n/helper';
 
class TMPackageDetails extends Component {
 
  render () {
    const {packageDetail, autoRenew, isConvergence, isPrepaid} = this.props;
    return (
      <ScrollView style={styles.scrollContainer} contentContainerStyle={styles.scrollContentContainer}>
        <WrapperCard>
          <HeaderWithCaret>
            <PackageDetailHeader isConvergence={isConvergence} name={translate('BILLS_USAGE_TRUEMOVE')} phoneNumber={packageDetail.phoneNumber}
              endMonth={packageDetail.endMonth} endDate={packageDetail.endDate} isPrepaid={isPrepaid}/>
          </HeaderWithCaret>
          <View style={styles.container}>
            <View style={styles.topContainer}>
              <PackageOverview title = {packageDetail.title} subTitle = {packageDetail.subtitle}/>
              {packageDetail.items && packageDetail.items.voice && <DetailCardRow rowData={packageDetail.items.voice} type='voice'/>}
              {packageDetail.items && packageDetail.items.data && <DetailCardRow rowData={packageDetail.items.data} type='data'/>}
              {packageDetail.items && packageDetail.items.wifi && <DetailCardRow rowData={packageDetail.items.wifi} type='wifi'/>}
              {packageDetail.items && packageDetail.items.sms && <DetailCardRow rowData={packageDetail.items.sms} type='sms' showSeparator={false}/>}
            </View>
            {!isPrepaid && <View>
              <View style={styles.creditContainer}>
                <View style={[styles.creditItem, styles.creditItemSeparator]}>
                  <ISText type='BOLD' style={styles.creditItemText}>{translate('PACKAGE_DETAIL__CREDIT_LIMIT')}{'\n'}<ISText type='BOLD' style={styles.creditItemCost}>{packageDetail.creditLimit}</ISText>{'\nBaht'}</ISText>
                </View>
                <View style={styles.creditItem}>
                  <ISText type='BOLD' style={styles.creditItemText}>{translate('PACKAGE_DETAIL__EXCEED_USAGE')}{'\n'}<ISText type='BOLD' style={styles.creditItemCost}>{packageDetail.exceedUsage}</ISText>{'\nBaht'}</ISText>
                </View>
              </View>
              <View style={styles.iconContainer}>
                <View style={styles.autoRenewContainer}>
                  <ISText type='BOLD' style={styles.autoRenewStatus}>{translate('PACKAGE_DETAIL__AUTO_RENEW')} : <ISText type='BOLD' style={styles.autoRenewStatusVal}>{autoRenew.isEnabled ? translate('PACKAGE_DETAIL__ON') : translate('PACKAGE_DETAIL__OFF')}</ISText></ISText>
                  <ISText><ISText type='BOLD' style={styles.autoRenewMethod}>{translate('PACKAGE_DETAIL__VIA_CC')} : </ISText><ISText type='BOLD'>{autoRenew.paymentDetail}</ISText></ISText>
                </View>
                <Icon style={styles.subIcon} size={20} name='chevron-right' />
              </View>
 
            </View>}
          </View>
        </WrapperCard>
      </ScrollView>
    );
  }
}
 
TMPackageDetails.defaultProps = {
  packageDetail: {},
  autoRenew: {},
  isConvergence: false,
  isPrepaid: false
};
 
TMPackageDetails.propTypes = {
  packageDetail: PropTypes.object,
  autoRenew: PropTypes.object,
  isConvergence: PropTypes.bool,
  isPrepaid: PropTypes.bool
};
 
export default TMPackageDetails;
