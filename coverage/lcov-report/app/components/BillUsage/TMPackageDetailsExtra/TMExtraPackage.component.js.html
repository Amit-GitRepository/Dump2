import HeaderWithCaret from '../BillUsage/BillUsageShared/HeaderWithCaret.component';
import ISText from '../../Shared/ISText/ISText.component';
import moment from '../../../utils/moment.util';
import partition from 'lodash/partition';
import ProgressBar from '../../Shared/ProgressBar/ProgressBar.component';
import PropTypes from 'prop-types';
import React, {Component} from 'react';
import styles from './TMExtraPackage.style';
import WrapperCard from '../BillUsage/BillUsageShared/WrapperCard.component';
import {colors} from '../../../themes/constants.styles';
import {ScrollView, View} from 'react-native';
import {translate} from '../../../language/i18n/helper';
 
class TMExtraPackage extends Component {
 
  render () {
    const {extraPackage} = this.props;
    const filteredPackages = partition(extraPackage, (item) => item.isActive);
    const rowContent = (packageRowItem, index) => (
      <View key={index} style={styles.packageItem}>
        <View style={styles.packageItemRow}>
          <View style={styles.packageRowLeft}>
            <ISText type='BOLD' style={styles.itemTitle}>{packageRowItem.title}<ISText type='BOLD' style={styles.itemSubtitle}> {packageRowItem.subtitle}</ISText></ISText>
          </View>
          <View style={styles.packageRowRight}>
            <ISText type='BOLD' style={styles.consumedValue}>{packageRowItem.consumedValue} <ISText style={styles.consumedValueUnit}>{packageRowItem.unit}</ISText></ISText>
          </View>
        </View>
        <View style={styles.packageItemRow}>
          <View style={styles.packageRowLeft}>
            <ProgressBar fillColor={colors.SECONDARY_TMOVE} backgroundColor={colors.PRIMARY_BG_SEPARATOR} percentage={(parseInt(packageRowItem.consumedValue) * 100) / parseInt(packageRowItem.totalValue)} />
          </View>
          <View style={styles.packageRowRight}>
            <ISText type='SEMIBOLD' style={styles.totalValue}>of {packageRowItem.totalValue} {packageRowItem.unit}</ISText>
          </View>
        </View>
        {packageRowItem.expiryDate && <ISText type='SEMIBOLD' style={styles.validityText}>Valid until {new moment(packageRowItem.expiryDate).format('DD/MM/YYYY')} </ISText>}
      </View>
    );
    const activePackagesContent = filteredPackages[0] && filteredPackages[0].map((packageRowItem, index) => rowContent(packageRowItem, index));
    const inactivePackagesContent = filteredPackages[1] && filteredPackages[1].map((packageRowItem, index) => rowContent(packageRowItem, index));
    return (
      <ScrollView style={styles.scrollContainer} contentContainerStyle={styles.scrollContentContainer}>
        <WrapperCard>
          <HeaderWithCaret headerColor={colors.PRIMARY_SUBMENU}>
            <ISText type='BOLD' style={styles.cardHeaderText}>{translate('EXTRA_PACKAGE_ACTIVE')}</ISText>
          </HeaderWithCaret>
          <View style={styles.cardContent}>
            {activePackagesContent}
          </View>
        </WrapperCard>
        <WrapperCard>
          <HeaderWithCaret headerColor={colors.PRIMARY_DISABLED_BG_TEXT}>
            <ISText type='BOLD' style={styles.cardHeaderText}>{translate('EXTRA_PACKAGE_INACTIVE')}</ISText>
          </HeaderWithCaret>
          <View style={styles.cardContent}>
            {inactivePackagesContent}
          </View>
        </WrapperCard>
      </ScrollView>
    );
  }
}
 
TMExtraPackage.defaultProps = {
  extraPackage: []
};
 
TMExtraPackage.propTypes = {
  extraPackage: PropTypes.array
};
 
export default TMExtraPackage;
