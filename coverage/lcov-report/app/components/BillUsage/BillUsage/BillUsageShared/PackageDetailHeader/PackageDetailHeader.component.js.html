import ISText from '../../../../Shared/ISText/ISText.component';
import Proptypes from 'prop-types';
import React, {Component} from 'react';
import styles from './PackageDetailHeader.style';
import {translate} from '../../../../../language/i18n/helper';
import {View} from 'react-native';
 
class PackageDetailHeader extends Component {
  render () {
    const {phoneNumber, name, endDate, endMonth, isConvergence, isPrepaid} = this.props;
    return (
      <View style = {(isConvergence || isPrepaid) ? styles.headerWrapperCenter : styles.headerWrapper}>
        <View style = {(isConvergence || isPrepaid) && styles.alignCenter}>
          <ISText type = 'BOLD' style = {styles.textWhite}>{name}</ISText>
          <ISText type = 'BOLD' style = {styles.textWhiteLarge}>{phoneNumber}</ISText>
        </View>
        { (!isConvergence && !isPrepaid) ?
          <View style = {styles.verticalLine}>
            <ISText type = 'BOLD' style = {styles.textWhite}>{translate('BILLS_USAGE_BILLING_CYCLE_END_DATE')}</ISText>
            <ISText style = {styles.textMonth}>{endMonth}<ISText type = 'BOLD' style = {styles.textDate}> {endDate}</ISText></ISText>
          </View> :
          null}
      </View>
    );
  }
}
PackageDetailHeader.defaultProps = {
  name: '',
  phoneNumber: '',
  endMonth: '',
  endDate: 1,
  isConvergence: false,
  isPrepaid: false
};
PackageDetailHeader.propTypes = {
  phoneNumber: Proptypes.string,
  name: Proptypes.string,
  endMonth: Proptypes.string,
  endDate: Proptypes.number,
  isConvergence: Proptypes.bool,
  isPrepaid: Proptypes.bool
};
export default PackageDetailHeader;
