import Icon from '../../../../Shared/Icon/Icon.component';
import noop from 'lodash/noop';
import Proptypes from 'prop-types';
import React, {Component} from 'react';
import styles from './CircularAccordianButton.style';
import Touchable from '../../../../Shared/Touchable/Touchable.component';
import {View} from 'react-native';
import * as Animatable from 'react-native-animatable';
 
const AnimatableIcon = Animatable.createAnimatableComponent(Icon);
 
class CircularAccordianButton extends Component {
  render () {
    const {isCollapsed, onClick, style} = this.props;
    const arrowStyle = !isCollapsed ? [styles.collapsedArrowStyle, styles.arrowStyle] : styles.arrowStyle;
    const circularContainerStyle = !isCollapsed ? [styles.circularContainer, styles.closedContainer] : styles.circularContainer;
    return (
      <Touchable onPress={onClick} style={[styles.container, style]}>
        <View style={circularContainerStyle}>
          <AnimatableIcon name={'chevron-down'} style={arrowStyle} transition='rotate' useNativeDriver/>
        </View>
      </Touchable>);
  }
}
CircularAccordianButton.defaultProps = {
  isCollapsed: true,
  onClick: noop,
  style: {}
};
CircularAccordianButton.propTypes = {
  isCollapsed: Proptypes.bool,
  onClick: Proptypes.func,
  style: Proptypes.object
};
 
export default CircularAccordianButton;
