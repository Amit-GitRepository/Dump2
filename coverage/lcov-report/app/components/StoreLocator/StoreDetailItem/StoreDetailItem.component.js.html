/* StoreDetailItem: show shop detail item such as distance, address etc. */
 
import ISText from '../../Shared/ISText/ISText.component';
import PropTypes from 'prop-types';
import React, {Component} from 'react';
import styles from './StoreDetailItem.style';
import {View} from 'react-native';
 
class StoreDetailItem extends Component {
  render () {
    const {heading, label, fontWeight} = this.props;
    if (!heading || !label) {
      return null;
    }
    return (
      <View style={styles.detailItem}>
        <ISText style={styles.detailText} type={fontWeight}>{this.props.heading}</ISText>
        <ISText style={styles.detailText} type={fontWeight}>{this.props.label}</ISText>
      </View>);
  }
}
StoreDetailItem.defaultProps = {
  heading: '',
  label: '',
  fontWeight: 'REGULAR'
};
StoreDetailItem.propTypes = {
  fontWeight: PropTypes.string,
  heading: PropTypes.oneOfType([PropTypes.string, PropTypes.number]),
  label: PropTypes.oneOfType([PropTypes.string, PropTypes.number])
};
export default StoreDetailItem;
